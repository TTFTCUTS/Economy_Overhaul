
# General labour
defined_text = {
    name = eco_labour_general
    value = modifier:eco_labour_general_add
}
defined_text = {
    name = eco_labour_general_demand
    value = modifier:eco_labour_general_demand_add
}

defined_text = {
    name = value_eco_planet_view_labour_general_summary_value
    value = value:eco_planet_view_labour_summary|type|general|
}

defined_text = {
    name = value_eco_planet_view_labour_general_summary
    random = no

    text = {
        trigger = {
            check_modifier_value = {
                modifier = eco_labour_general_demand_add
                value <= 0
            }
        }
        localization_key = eco_planet_view_labour_summary_na
    }

    text = {
        trigger = {
            check_variable_arithmetic = {
                which = value:eco_planet_labour_fulfillment_uncapped|type|general|
                value < 0.5
            }
        }
        localization_key = eco_planet_view_labour_general_summary_extremely_low
    }

    text = {
        trigger = {
            check_variable_arithmetic = {
                which = value:eco_planet_labour_fulfillment_uncapped|type|general|
                value < 0.75
            }
        }
        localization_key = eco_planet_view_labour_general_summary_very_low
    }

    text = {
        trigger = {
            check_variable_arithmetic = {
                which = value:eco_planet_labour_fulfillment_uncapped|type|general|
                value < 1.0
            }
        }
        localization_key = eco_planet_view_labour_general_summary_low
    }

    text = {
        trigger = {
            check_variable_arithmetic = {
                which = value:eco_planet_labour_fulfillment_uncapped|type|general|
                value >= 1.2
            }
        }
        localization_key = eco_planet_view_labour_general_summary_high
    }

    default = eco_planet_view_labour_general_summary_normal
}
